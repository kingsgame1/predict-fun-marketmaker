/**
 * Market Maker Configuration Module
 * Handles all trading-related configuration
 */
 
import { parseBool, parseNumber, parseDuration } from './parsers.js';
 
export interface TradingConfig {
  enableTrading: boolean;
  spread: number;
  minSpread: number;
  maxSpread: number;
  orderSize: number;
  maxPosition: number;
  maxDailyLoss: number;
  minIntervalMs: number;
  cooldownAfterCancelMs: number;
}
 
export interface MarketMakerConfig {
  // Spread settings
  mmSpreadVolWeight?: number;
  mmSpreadLiquidityWeight?: number;
  mmBookSpreadWeight?: number;
  mmPressureSpreadWeight?: number;
  mmTouchBufferBps?: number;
  mmTouchBufferVolWeight?: number;
 
  // Depth settings
  mmDepthMinShares?: number;
  mmDepthRefShares?: number;
  mmDepthTargetShares?: number;
  mmDepthPenaltyWeight?: number;
 
  // Volatility settings
  mmVolatilityThresholdBps?: number;
  mmVolatilityCooldownMs?: number;
  mmSpreadJumpBps?: number;
  mmProtectiveSpreadJumpBps?: number;
 
  // Near-touch and anti-fill settings
  nearTouchBps?: number;
  antiFillBps?: number;
  mmNearTouchVolMultiplier?: number;
  mmNearTouchDepthSpeedBps?: number;
  mmAntiFillVolMultiplier?: number;
  mmAntiFillDepthSpeedMult?: number;
 
  // Safety settings
  mmSafeModeSpreadMult?: number;
  mmSafeModeSpreadAdd?: number;
  mmSafeModePauseMs?: number;
  mmSafeModeNearTouchMult?: number;
  mmSafeModeHardCancelMult?: number;
 
  // Dynamic spread settings
  mmDynamicSpreadEnabled?: boolean;
  mmDynamicSpreadMinConfidence?: number;
 
  // Anti-fill strategy settings
  mmAntiFillEnabled?: boolean;
  mmAntiFillNearTouchBps?: number;
  mmAntiFillImbalanceThreshold?: number;
  mmAntiFillPressureThreshold?: number;
  mmAntiFillDepthSpeedBps?: number;
  mmAntiFillMinCancelIntervalMs?: number;
}
 
/**
 * Load trading configuration from environment
 */
export function loadTradingConfig(): TradingConfig {
  return {
    enableTrading: parseBool(process.env.ENABLE_TRADING, false),
    spread: parseNumber(process.env.SPREAD, 0.02) ?? 0.02,
    minSpread: parseNumber(process.env.MIN_SPREAD, 0.01) ?? 0.01,
    maxSpread: parseNumber(process.env.MAX_SPREAD, 0.08) ?? 0.08,
    orderSize: parseNumber(process.env.ORDER_SIZE, 25) ?? 25,
    maxPosition: parseNumber(process.env.MAX_POSITION, 100) ?? 100,
    maxDailyLoss: parseNumber(process.env.MAX_DAILY_LOSS, 200) ?? 200,
    minIntervalMs: parseDuration(process.env.MIN_INTERVAL_MS) ?? 3000,
    cooldownAfterCancelMs: parseDuration(process.env.COOLDOWN_AFTER_CANCEL_MS) ?? 4000,
  };
}

/**
 * Validate trading configuration
 */
export function validateTradingConfig(config: TradingConfig): string[] {
  const errors: string[] = [];
 
  if (config.spread <= 0) {
    errors.push('SPREAD must be positive');
  }
 
  if (config.minSpread > config.maxSpread) {
    errors.push('MIN_SPREAD cannot be greater than MAX_SPREAD');
  }
 
  if (config.orderSize <= 0) {
    errors.push('ORDER_SIZE must be positive');
  }
 
  if (config.maxPosition <= 0) {
    errors.push('MAX_POSITION must be positive');
  }
 
  return errors;
}
